If you dont want to use a plugin for changing your excerpt style you can copy/paste those functions you need and to put them in your themes functions.php file.

// Changing excerpt length (default 55 words)
function new_excerpt_length($length) {
	return 100;
}
add_filter('excerpt_length', 'new_excerpt_length');


// Changing excerpt 'more' text (default […])
function new_excerpt_more($more) {
	return ' (Continue reading…)';
}
add_filter('excerpt_more', 'new_excerpt_more');


// Changing excerpt more
function new_excerpt_more($more) {
	global $post;
	return '… <a href="'. get_permalink($post->ID) . '">' . 'Read More &raquo;' . '</a>';
}
add_filter('excerpt_more', 'new_excerpt_more');
